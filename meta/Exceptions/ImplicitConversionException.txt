                    Int --> Double
                    Int --> Bool
                    Int --> Char
                    Char --> Int
                    Double --> Int
                    Bool --> Int
                    String --> String
                    List --> List
                    FileReader --> FileReader
                    FileWriter --> FileWriter
                    LiteralString --> String
